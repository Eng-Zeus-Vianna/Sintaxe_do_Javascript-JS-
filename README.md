"elementos" especiais, como <head>, 
  <title>, 
    <body>, 
      <header>, 
        <footer>, 
          <article>, 
            <section>,
              <p>, 
              <div>, 
                <span>, 
                  <img>, 
                  <aside>, 
                    <audio>,
                      <canvas>, 
                        <datalist>, 
                          <details>, 
                            <embed>, 
                            <nav>, 
                            <output>, 
                              <progress>, 
                                <video>, 
                                  <ul>, 
                                    <ol>, 
                                      <li> e muitos outros.
